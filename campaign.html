/* ========================================================================== ЛОГИКА АВТО-ПОХОДОВ ДЛЯ МАКСИМУМ 360 МИНУТ В СУТКИ ========================================================================== */ /* Загружаем статы персонажа и состояние походов */ let heroStats = JSON.parse(localStorage.getItem('hero_stats')) || { name: "Воин", silver: 1000, gold: 0, level: 1 }; let campState = JSON.parse(localStorage.getItem('campaign_state')) || { isEnRoute: false, returnTime: 0, lastDuration: 0, lastSilverReward: 0, lastGoldReward: 0, timeUsedToday: 0, lastResetDate: "" }; /* Функция проверки и сброса суточного таймера на 360 минут */ function checkDailyReset() { const todayStr = new Date().toDateString(); if (campState.lastResetDate !== todayStr) { campState.timeUsedToday = 0; /* Сбрасываем потраченные минуты */ campState.lastResetDate = todayStr; localStorage.setItem('campaign_state', JSON.stringify(campState)); } } /* Основной цикл обновления интерфейса */ function updateCampaignUI() { checkDailyReset(); const box = document.getElementById('campaign-logic-box'); if (!box) return; const now = Date.now(); const timeLeftToday = 360 - campState.timeUsedToday; /* Сценарий 1: Персонаж прямо сейчас идет по лесам */ if (campState.isEnRoute) { if (now >= campState.returnTime) { /* Время вышло — поход завершен */ processCampaignEnd(); return; } /* Высчитываем сколько секунд осталось до возвращения коня */ let secondsLeft = Math.ceil((campState.returnTime - now) / 1000); let displayMins = Math.floor(secondsLeft / 60); let displaySecs = secondsLeft % 60; if (displaySecs < 10) displaySecs = "0" + displaySecs; box.innerHTML = `
🛡️ Ваша дружина в походе!
Ваш герой вернется через: ${displayMins}:${displaySecs}
Вы собираете трофеи на землях врагов...
`; /* Запускаем ежесекундный тик */ setTimeout(updateCampaignUI, 1000); return; } /* Сценарий 2: Персонаж дома, выводим результаты прошлого рейда и меню отправки */ let lastResultHTML = ""; if (campState.lastDuration > 0) { let goldText = campState.lastGoldReward > 0 ? `, 👑 ${campState.lastGoldReward} зол.` : ""; lastResultHTML = `
Последний поход:
Ты был в походе ${campState.lastDuration} минут, награда: ⚪ ${campState.lastSilverReward} сер.${goldText}
`; } box.innerHTML = ` ${lastResultHTML}
Осталось времени на поход: ${timeLeftToday} минут
${timeLeftToday <= 0 ? `
Вы исчерпали суточный лимит в 360 минут! Возвращайтесь завтра.
` : `
` } `; } /* Действие: Отправка коня в поход */ window.startCampaignRequest = function() { const select = document.getElementById('camp-duration-select'); if (!select) return; const mins = parseInt(select.value); const timeLeftToday = 360 - campState.timeUsedToday; if (mins > timeLeftToday) { alert("У вас нет столько доступного времени на сегодня!"); return; } /* Для теста и удобства разработки переводим 1 минуту похода в 1 секунду ожидания */ /* (Если нужно строго реальное время минут, замените "* 1000" на "* 60 * 1000") */ const durationInMs = mins * 1000; campState.isEnRoute = true; campState.returnTime = Date.now() + durationInMs; campState.lastDuration = mins; campState.timeUsedToday += mins; localStorage.setItem('campaign_state', JSON.stringify(campState)); updateCampaignUI(); }; /* Логика: Начисление наград при возвращении */ function processCampaignEnd() { const mins = campState.lastDuration; /* Базовый расчет серебра на основе минут */ let silverEarned = Math.floor(mins * 70 + Math.random() * 50); let goldEarned = 0; /* Логика золота: за 60-минутный поход даём шанс выбить до 1 золотого */ if (mins === 60) { if (Math.random() < 0.8) goldEarned = 1; /* Высокий 80% шанс принести 1 золото */ } else if (mins === 30) { if (Math.random() < 0.3) goldEarned = 1; /* 30% шанс для тридцатиминутки */ } /* Обновляем кошелек героя */ heroStats.silver += silverEarned; heroStats.gold += goldEarned; /* Обновляем состояние модуля */ campState.isEnRoute = false; campState.returnTime = 0; campState.lastSilverReward = silverEarned; campState.lastGoldReward = goldEarned; localStorage.setItem('hero_stats', JSON.stringify(heroStats)); localStorage.setItem('campaign_state', JSON.stringify(campState)); updateCampaignUI(); if (typeof updateHeader === 'function') updateHeader(); /* Синхронизируем шапку */ } /* Инициализация модуля при загрузке */ updateCampaignUI();
